\subsubsection*{Celsius to Farenheit}
$x\degree F=(y\degree C*(9/5))+32$
%-------------------------------------------------------------------------------
\subsubsection*{Farenheit to Celsius}
$y\degree C=(x\degree F-32)*(5/9)$
